import torch
import torch.nn as nn
# #Single variable regression by torch
#y=2*x
x=torch.tensor([[0.1],[0.2],[0.3],[0.4]], dtype=torch.float32)
y=torch.tensor([[0.2],[0.4],[0.6],[0.8]], dtype=torch.float32)
#w=torch.tensor(0.0, dtype=torch.float32,requires_grad=True)

n_sample, n_feature = x.shape

input_size = 1
output_size = 1

#Using classes to define nn

class LinearRegression(torch.nn.Module):
    def __init__(self,input_dim,output_dim):
        super().__init__()
        #Define the Layers
        self.lin= nn.Linear(input_dim,output_dim)
        
    def forward(self,x: torch.Tensor):
        return self.lin(x)
        
    
        
model = LinearRegression(input_size,output_size)    

#learning rate
lr=0.1

n_iter = 1000
#Loss Function
loss_nn = torch.nn.MSELoss() #Mean Square Error
optimizer = torch.optim.SGD(model.parameters(),lr=0.8) #Stochastic Gradient Descent



for epoch in range(n_iter):
    
    #forward pass
    y_pred =model(x)
    
    #loss function
    l=loss_nn(y_pred,y)
    
    optimizer.zero_grad()
    #Compute gradients
    
    l.backward() #sets loss funciton as the function which we are computing partials for
    
    #Update w
    optimizer.step() #Take 1 step of gradient descent
    
    [w,b]=model.parameters()
    if epoch % 100 == 0:
        print(f'epoch {epoch}: w={w.item():.3f}, b={b.item():.3f}  loss ={l:.3f}')

import sklearn
import warnings

import pandas as pd
dataset = load_boston()
df=pd.DataFrame(dataset.data)
df.columns=dataset.feature_names
df["Price"]=dataset.target 
#house price is in the last column, there might be a warning message

#Set our target as the housing prices
Y = df["Price"]
#Make our features by dropping our target column
X = df.drop(['Price'],axis = 1)

#Reshape appropriately as torch tensors
Y = torch.Tensor(Y).reshape(-1,1)
X = torch.Tensor(X.values)

#Split data
X_train, X_test, Y_train, Y_test = train_test_split(X,Y,test_size= 0.25)

#Scale data
x_scaler = sklearn.preprocessing.StandardScaler()
X_train = x_scaler.fit_transform(X_train)
X_test = x_scaler.transform(X_test)

#I think the scaling switched our data back to a numpy array. Switch back to tensor (oops)
X_train =torch.from_numpy(X_train.astype(np.float32))
X_test = torch.from_numpy(X_test.astype(np.float32))

#Construct our Network
n_features = X_train.shape[1]
class LinearRegression(nn.Module):
    def __init__(self, n_features):
        super(LinearRegression,self).__init__()
        #Define the Layers
        self.network=nn.Sequential(
        nn.Linear(n_features,40),

        nn.Linear(40,20),
        nn.ReLU(),
        nn.Linear(20,10),
        nn.ReLU(),
        nn.Linear(10,1),
        nn.ReLU()
        )
        
    def forward(self,x):
